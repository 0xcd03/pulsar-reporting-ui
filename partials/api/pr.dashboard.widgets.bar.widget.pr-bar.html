pr-bar
object in module pr.dashboard.widgets.bar

Description

Widget for nvd3 discrete bar chart. This widget displays data in a chart for a single metric and a single dimension.

Usage:

HTML:

<pr-dashboard-widget widget="widget"></pr-dashboard-widget>

JS:

$scope.widget = {
  "type": "pr-bar",
  "params": {
    "dimensions": [
      {
        "name": "browserfamily"
      }
    ],
    "metrics": [
      {
        "name": "clickcount_ag",
        "type": "count",
        "alias": "clickcount_ag"
      }
    ],
    "maxResults": 100
    "granularity": "all",
    "dataSourceName": "pulsar_ogmb"
  },
  "options": {
    "title": "A bar chart widget",
    "disabled": false,
    "chart" : {
       // Additional options that will be sent to nvd3 chart, example available at http://nvd3.org/examples/discreteBar.html
    }
  }
}
